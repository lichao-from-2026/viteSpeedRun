### 1.1 自定义打包逻辑

#### 1.1.1 理解 Rollup 构建过程

Vite 的生产构建基于 Rollup，通过插件可以自定义构建过程：

**插件代码：**

```typescript
export function customBuildPlugin() {
  return {
    name: 'vite-plugin-custom-build',
    
    // 构建开始时执行
    buildStart() {
      console.log('Custom build started!')
    },
    
    // 自定义 chunk 渲染
    renderChunk(code, chunk) {
      // 对特定 chunk 进行处理
      if (chunk.name === 'main') {
        // 添加构建时间戳
        const timestamp = new Date().toISOString()
        return {
          code: code + `\nconsole.log('Built at: ${timestamp}')`,
          map: null
        }
      }
      return null
    },
    
    // 构建完成时执行
    buildEnd(error) {
      if (error) {
        console.error('Build failed:', error)
      } else {
        console.log('Custom build completed!')
      }
    }
  }
}
```

#### 1.1.2 自定义输出格式

**插件代码：**

```typescript
export function customOutputPlugin() {
  return {
    name: 'vite-plugin-custom-output',

    config(config) {

  }
}
```

### 1.2 集成第三方工具

#### 1.2.1 集成 ESLint

**插件代码：**

```typescript
import { ESLint } from 'eslint'

export function eslintPlugin(options = {}) {
  const { fix = false, include = '**/*.{js,ts,jsx,tsx}' } = options
  
  return {
    name: 'vite-plugin-eslint',
    
    async transform(code, id) {
      // 只处理指定文件
      if (id.match(include)) {
        const eslint = new ESLint({
          fix,
          useEslintrc: true
        })
        
        try {
          const results = await eslint.lintText(code, { filePath: id })
          
          if (fix) {
            await ESLint.outputFixes(results)
          }
          
          // 输出错误信息
          results.forEach(result => {
            result.messages.forEach(message => {
              if (message.severity === 2) {
                console.error(`ESLint error in ${id}:${message.line}:${message.column}`)
                console.error(`  ${message.message}`)
              }
            })
          })
        } catch (error) {
          console.error('ESLint error:', error)
        }
      }
      
      return null
    }
  }
}
```

#### 1.2.2 集成 Prettier

**插件代码：**

```typescript
import prettier from 'prettier'

export function prettierPlugin(options = {}) {
  const { configFile = '.prettierrc' } = options
  
  return {
    name: 'vite-plugin-prettier',
    
    async transform(code, id) {
      // 只处理 JavaScript 和 TypeScript 文件
      if (id.match(/\.(js|ts|jsx|tsx)$/)) {
        try {
          const formattedCode = await prettier.format(code, {
            configFile,
            filepath: id
          })
          
          if (formattedCode !== code) {
            return {
              code: formattedCode,
              map: null
            }
          }
        } catch (error) {
          console.error('Prettier error:', error)
        }
      }
      
      return null
    }
  }
}
```

### 1.3 实现自定义热更新

#### 1.3.1 自定义热更新逻辑

**插件代码：**

```typescript
export function customHmrPlugin() {
  return {
    name: 'vite-plugin-custom-hmr',
    
    handleHotUpdate(ctx) {
      // 处理特定文件的热更新
      if (ctx.file.endsWith('.json')) {
        console.log('JSON file changed:', ctx.file)
        
        // 重新加载所有依赖该 JSON 文件的模块
        const modules = ctx.modules.filter(module => 
          module.id.includes('src')
        )
        
        return modules
      }
      
      return ctx.modules
    }
  }
}
```

#### 1.3.2 热更新优化

**插件代码：**

```typescript
export function hmrOptimizationPlugin() {
  return {
    name: 'vite-plugin-hmr-optimization',
    
    handleHotUpdate(ctx) {
      // 分析文件变化影响的模块
      const affectedModules = new Set()
      
      // 收集直接依赖
      ctx.modules.forEach(module => {
        affectedModules.add(module)
        
        // 收集间接依赖
        module.importedModules.forEach(importedModule => {
          affectedModules.add(importedModule)
        })
      })
      
      console.log('Affected modules:', Array.from(affectedModules).map(m => m.id))
      
      // 只更新受影响的模块
      return Array.from(affectedModules)
    }
  }
}
```

## 三、实操：开发一个用于业务场景的插件

### 3.1 自动生成接口请求文件

#### 3.1.1 插件功能说明

创建一个插件，根据 API 文档自动生成接口请求文件。

#### 3.1.2 插件实现

**插件代码：**

```typescript
import fs from 'fs'
import path from 'path'

export function apiGeneratorPlugin(options = {}) {
  const {
    apiDocsPath = 'api-docs.json',
    outputDir = 'src/api'
  } = options
  
  return {
    name: 'vite-plugin-api-generator',
    
    buildStart() {
      // 读取 API 文档
      if (fs.existsSync(apiDocsPath)) {
        const apiDocs = JSON.parse(fs.readFileSync(apiDocsPath, 'utf-8'))
        
        // 创建输出目录
        if (!fs.existsSync(outputDir)) {
          fs.mkdirSync(outputDir, { recursive: true })
        }
        
        // 生成 API 文件
        const apiCode = generateApiCode(apiDocs)
        fs.writeFileSync(path.join(outputDir, 'index.ts'), apiCode)
        
        console.log('API files generated successfully!')
      } else {
        console.warn(`API docs not found at ${apiDocsPath}`)
      }
    }
  }
}

function generateApiCode(apiDocs) {
  let code = 'import axios from "axios"\n\n'
  
  // 生成 API 函数
  Object.entries(apiDocs).forEach(([name, config]) => {
    code += `export function ${name}(${config.params ? config.params : ''}) {\n`
    code += `  return axios({\n`
    code += `    url: "${config.url}",\n`
    code += `    method: "${config.method || 'GET'}",\n`
    if (config.params) {
      code += `    params,\n`
    }
    code += `  })\n`
    code += `}\n\n`
  })
  
  return code
}
```

#### 3.1.3 API 文档格式

**api-docs.json：**

```json
{
  "getUsers": {
    "url": "/api/users",
    "method": "GET",
    "params": "params"
  },
  "createUser": {
    "url": "/api/users",
    "method": "POST",
    "params": "userData"
  }
}
```

#### 3.1.4 使用示例

**在 vite.config.ts 中：**

```typescript
import { defineConfig } from 'vite'
import { apiGeneratorPlugin } from './plugins/api-generator'

export default defineConfig({
  plugins: [
    apiGeneratorPlugin({
      apiDocsPath: 'api-docs.json',
      outputDir: 'src/api'
    })
  ]
})
```

**在代码中使用：**

```typescript
import { getUsers, createUser } from './api'

// 使用生成的 API 函数
async function fetchUsers() {
  const response = await getUsers({ page: 1, limit: 10 })
  console.log(response.data)
}

async function addUser() {
  const response = await createUser({ name: 'John', email: 'john@example.com' })
  console.log(response.data)
}
```

### 3.2 处理后端返回的 Mock 数据

#### 3.2.1 插件功能说明

创建一个插件，根据后端返回的 Mock 数据生成 Mock 服务。

#### 3.2.2 插件实现

**插件代码：**

```typescript
import fs from 'fs'
import path from 'path'

export function mockDataPlugin(options = {}) {
  const {
    mockDataPath = 'mock-data.json',
    mockApiPrefix = '/api'
  } = options
  
  let mockData = {}
  
  return {
    name: 'vite-plugin-mock-data',
    
    configResolved(resolvedConfig) {
      // 只在开发环境启用
      if (resolvedConfig.command === 'serve') {
        // 读取 Mock 数据
        if (fs.existsSync(mockDataPath)) {
          mockData = JSON.parse(fs.readFileSync(mockDataPath, 'utf-8'))
          console.log('Mock data loaded successfully!')
        } else {
          console.warn(`Mock data not found at ${mockDataPath}`)
        }
      }
    },
    
    configureServer(server) {
      // 添加 Mock 中间件
      server.middlewares.use((req, res, next) => {
        // 只处理 API 请求
        if (req.url.startsWith(mockApiPrefix)) {
          const apiPath = req.url.replace(mockApiPrefix, '')
          
          // 查找对应的 Mock 数据
          const mockResponse = mockData[apiPath]
          
          if (mockResponse) {
            res.setHeader('Content-Type', 'application/json')
            res.end(JSON.stringify(mockResponse))
            return
          }
        }
        
        next()
      })
    }
  }
}
```

#### 3.2.3 Mock 数据格式

**mock-data.json：**

```json
{
  "/users": [
    { "id": 1, "name": "John", "email": "john@example.com" },
    { "id": 2, "name": "Jane", "email": "jane@example.com" }
  ],
  "/users/1": {
    "id": 1, "name": "John", "email": "john@example.com", "age": 30
  }
}
```

#### 3.2.4 使用示例

**在 vite.config.ts 中：**

```typescript
import { defineConfig } from 'vite'
import { mockDataPlugin } from './plugins/mock-data'

export default defineConfig({
  plugins: [
    mockDataPlugin({
      mockDataPath: 'mock-data.json',
      mockApiPrefix: '/api'
    })
  ]
})
```

**在代码中使用：**

```typescript
// 直接请求 API，会返回 Mock 数据
async function fetchUsers() {
  const response = await fetch('/api/users')
  const users = await response.json()
  console.log(users)
}

async function fetchUser(id) {
  const response = await fetch(`/api/users/${id}`)
  const user = await response.json()
  console.log(user)
}
```
