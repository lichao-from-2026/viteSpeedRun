# 核心配置

## 0. 完整配置示例

以下是一个包含上述所有核心配置的完整示例：

```typescript
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'
import path from 'path'

export default defineConfig({
  base: '/',                                          // 项目基础路径

  plugins: [react()],                                 // 插件配置

  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),         // 指向 src 目录
      '@components': path.resolve(__dirname, './src/components'),  // 指向组件目录
      '@utils': path.resolve(__dirname, './src/utils'),         // 指向工具函数目录
      '@styles': path.resolve(__dirname, './src/styles')         // 指向样式目录
    },
    extensions: ['.tsx', '.ts', '.jsx', '.js', '.json']  // 可以省略的文件扩展名
  },

  server: {
    port: 5174,                                      // 开发服务器端口
    host: '0.0.0.0',                                // 主机地址，0.0.0.0 允许外部访问
    open: true,                                       // 启动时自动打开浏览器
    strictPort: false,                                // 端口被占用时是否直接退出
    cors: true,                                       // 启用 CORS，允许所有来源
    proxy: {
      '/api': {
        target: 'http://localhost:3000',              // 目标服务器地址
        changeOrigin: true,                             // 修改请求头中的 origin 为目标地址
        rewrite: (path) => path.replace(/^\/api/, '')  // 路径重写，移除 /api 前缀
      },
      '/upload': {
        target: 'http://localhost:4000',              // 上传服务地址
        changeOrigin: true,                             // 修改请求头中的 origin 为目标地址
        ws: true                                        // 支持 WebSocket
      }
    }
  },

  build: {
    outDir: 'dist',                                    // 输出目录
    assetsDir: 'assets',                               // 静态资源目录
    sourcemap: true,                                   // 是否生成 sourcemap
    minify: 'esbuild',                                // 代码压缩工具
    target: ['es2020', 'chrome87', 'firefox78', 'safari14'],  // 目标浏览器兼容性
    assetsInlineLimit: 4096                           // 资源内联限制（字节）
  },

  css: {
    modules: {
      localsConvention: 'camelCase',                 // CSS 类名转换方式（如 my-button → myButton）
      scopeBehaviour: 'local'                         // 默认使用局部作用域，避免样式污染
    },
    preprocessorOptions: {
      less: {
        math: 'always',                              // 始终启用数学表达式计算
        globalVars: {
          primaryColor: '#1890ff',                // 全局主色调变量
          secondaryColor: '#52c41a'               // 全局辅助色变量
        },
        additionalData: `@import "@styles/mixins.less;`  // 每个 Less 文件自动引入混合器
      },
      scss: {
        additionalData: `@import "@styles/variables.scss";`,  // 每个 SCSS 文件自动引入变量
        api: 'modern-compiler',                        // 使用现代 Sass 编译器
        includePaths: ['node_modules']                 // 允许从 node_modules 导入
      }
    },
    codeSplit: true                                    // 启用 CSS 代码分割，与 JS 路由同步加载
  },

  build: {
    outDir: 'dist',                                    // 输出目录
    assetsDir: 'assets',                               // 静态资源目录
    sourcemap: true,                                   // 是否生成 sourcemap
    minify: 'esbuild',                                // 代码压缩工具
    target: ['es2020', 'chrome87', 'firefox78', 'safari14'],  // 目标浏览器兼容性
    assetsInlineLimit: 4096                           // 小于 4KB 的资源内联为 base64
  },

  publicDir: 'public'                                 // 公共资源目录，文件原样复制到输出目录
})
```

## 1. server - 开发服务器配置

### 1.1 基础配置

**配置说明：** 开发服务器的核心配置项，包括端口、主机、自动打开浏览器等。

**常用配置项：**
- `port` - 开发服务器端口
- `host` - 指定主机地址
- `open` - 启动时自动打开浏览器
- `strictPort` - 端口被占用时是否直接退出
- `https` - 是否启用 HTTPS

**案例：**

```typescript
server: {
  port: 5174,              // 开发服务器端口
  host: '0.0.0.0',        // 主机地址，0.0.0.0 允许外部访问
  open: true,             // 启动时自动打开浏览器
  strictPort: false,      // 端口被占用时是否直接退出
  https: false            // 是否启用 HTTPS
}
```

---

### 1.2 代理配置（proxy）
> **使用思路：** 前端使用相对路径 `/api/xxx`，开发环境 请求Vite代理服务器转发请求，生成环境 直接请求正式服务器。

> **开发环境 请求流程：** 浏览器 → Vite 开发服务器（同域）→ 后端服务器（跨域）
>
> **流程说明：** 前端使用相对路径 `/api/xxx`，无需修改代码即可适配不同环境
> - 前端 请求 Vite代理服务器 `http://localhost:5174/api/users`（同域，浏览器不拦截）
> - Vite 代理检测到 `/api` 前缀，转发请求到后端服务：`http://localhost:3000/users`
> - 后端 处理请求并返回响应
> - Vite 将响应返回给浏览器

> **生产环境 请求流程：** 浏览器 → 后端服务器（直接请求）
>
> **流程说明：** 生产环境无需 Vite 代理，直接由 Nginx、CDN 或后端服务处理
> - 生产环境 前端代码已打包构建
> - 前端 请求 正式接口：`https://api.example.com/users`（或同域下的 `/api/users`）
> - 后端 处理请求并返回响应

**配置说明：** 开发环境中代理 API 请求，解决跨域问题。

**常用配置：**
- 简单代理
- 路径重写
- 多代理配置

**案例 1 - 简单代理：**

```typescript
server: {
  proxy: {
    '/api': {
      target: 'http://localhost:3000',
      changeOrigin: true
    }
  }
}
```

**案例 2 - 路径重写：**

```typescript
server: {
  proxy: {
    '/api': {
      target: 'http://localhost:3000',
      changeOrigin: true,
      rewrite: (path) => path.replace(/^\/api/, '')
    }
  }
}
```

**案例 3 - 多代理配置：**

```typescript
server: {
  proxy: {
    '/api': {
      target: 'http://localhost:3000',
      changeOrigin: true
    },
    '/upload': {
      target: 'http://localhost:4000',
      changeOrigin: true,
      ws: true
    }
  }
}
```

### 1.3 CORS 配置

> **Vite 中 `server.cors: true` 的作用：**
> - 允许开发服务器接收来自不同源的请求
> - 在大多数情况下，开发阶段直接访问 `localhost:5174` 时，CORS 配置的影响不大
> - CORS 配置解决的是：其他来源访问 Vite 开发服务器时的跨域问题
> 
> **需要访问 Vite 开发服务器资源的场景：**
> - 移动端开发（手机通过局域网访问电脑）
> - 微前端架构（主应用加载子应用）
> - 多项目联调（不同端口的前端互相访问）
> - 后端渲染页面嵌入前端资源
> - 局域网内同事预览你的开发效果

**配置说明：** 配置开发服务器的 CORS（跨域资源共享）。

**案例：**

```typescript
server: {
  cors: true
}
```

或自定义 CORS 配置：

```typescript
server: {
  cors: {
    origin: ['http://localhost:5174', 'http://localhost:3000'],
    credentials: true
  }
}
```

## 4. css - CSS 配置

### 【前置知识】CSS Modules 模式介绍和使用

#### CSS Modules 是什么？
CSS Modules 是一种 CSS 模块化解决方案，让 CSS 类名和动画名默认具有局部作用域，避免样式污染。

#### 使用方式
1. **文件名格式**：使用 `.module.css`、`.module.scss`、`.module.less` 等后缀
2. **导入方式**：`import styles from './Button.module.css'`
3. **使用方式**：`<div className={styles.container}></div>`

#### 优势
- **作用域隔离**：每个组件的样式互不影响
- **自动生成唯一类名**：避免类名冲突
- **类型安全**：结合 TypeScript 可以获得类型提示

#### 示例代码
```css
/* Button.module.css */
.container { padding: 10px; }
```

```typescript
/* Button.tsx */
import styles from './Button.module.css'
function Button() {
  return <div className={styles.container}>按钮</div>
}
```

---

### 4.1 CSS Modules 配置

**配置说明：** 配置 CSS Modules 的行为，实现样式作用域隔离和自动类名转换。

**常用配置项：**

- **localsConvention** - CSS 类名转换方式
  - `'camelCase'` - 将连字符类名转换为驼峰式（如 `my-button` → `myButton`）
  - `'camelCaseOnly'` - 只允许驼峰式类名，禁止连字符类名
  - `null` - 保持原样，不进行转换
  - **使用场景：** 提高 TypeScript 开发体验，避免连字符类名在 JS/TS 中需要用引号包裹

- **scopeBehaviour** - 作用域行为
  - `'local'` - 默认值，样式默认局部作用域，需要使用 `:global()` 定义全局样式
  - `'global'` - 样式默认全局作用域，需要使用 `:local()` 定义局部样式
  - **使用场景：** 根据项目需求调整默认作用域，大型项目推荐使用 `'local'`

**案例：**

```typescript
css: {
  modules: {
    localsConvention: 'camelCase',      // 类名转换为驼峰式（如 my-button → myButton）
    scopeBehaviour: 'local'               // 默认使用局部作用域，避免样式污染
  }
}
```

**使用示例：**

```css
/* button.module.css */
.my-button {
  background: blue;
}

:global(.global-button) {
  background: red;
}
```

```typescript
import styles from './button.module.css'

function Button() {
  return (
    <>
      <button className={styles.myButton}>局部样式按钮</button>
      <button className="global-button">全局样式按钮</button>
    </>
  )
}
```

---

### 4.2 预处理器配置

**配置说明：** 配置 CSS 预处理器（如 Less、Sass、Stylus）的选项，在 Vite 中无需额外安装预处理器插件，直接通过此配置即可使用。

**常用配置项说明：**

- **preprocessorOptions.less** - Less 预处理器配置
  - **math** - 数学表达式计算方式
    - `'always'` - 总是计算数学表达式（如 `100px - 20px`）
    - `'parens-division'` - 仅在括号中计算除法
    - `'strict'` - 严格模式，需要括号才能计算
    - `'on'` - 开启数学计算（默认值）
  - **globalVars** - 全局 Less 变量
    - 可以在任意 Less 文件中直接使用这些变量，无需单独引入
    - **使用场景：** 定义主题色、间距等全局样式变量
  - **javascriptEnabled** - 是否允许在 Less 中使用 JavaScript
    - 默认值：`false`
    - 安全考虑，一般不建议开启
  - **additionalData** - 在每个 Less 文件开头自动添加的内容
    - 可以用于自动引入全局变量文件

- **preprocessorOptions.scss/preprocessorOptions.sass** - Sass/SCSS 预处理器配置
  - **additionalData** - 在每个 SCSS/Sass 文件开头自动添加的内容
    - 可以用于自动引入全局变量、混合器文件
    - **使用场景：** 避免在每个文件中重复 import 变量文件
  - **api** - Sass 编译器 API 版本
    - `'modern-compiler'` - 使用现代 Sass 编译器（推荐，性能更好）
    - `'legacy'` - 使用旧版 Node Sass
  - **includePaths** - 额外的导入路径
    - 方便从 node_modules 或其他目录导入 Sass 文件

- **preprocessorOptions.stylus** - Stylus 预处理器配置

**案例 - Less：**

```typescript
css: {
  preprocessorOptions: {
    less: {
      math: 'always',                                    // 始终启用数学表达式计算
      globalVars: {
        primaryColor: '#1890ff',                        // 全局主色调变量
        secondaryColor: '#52c41a'                       // 全局辅助色变量
      },
      additionalData: `@import "@styles/mixins.less";`  // 每个 Less 文件自动引入混合器
    }
  }
}
```

**案例 - Sass：**

```typescript
css: {
  preprocessorOptions: {
    scss: {
      additionalData: `@import "@styles/variables.scss";`,  // 每个 SCSS 文件自动引入变量
      api: 'modern-compiler',                                  // 使用现代 Sass 编译器
      includePaths: ['node_modules']                           // 允许从 node_modules 导入
    }
  }
}
```

### 4.3 CSS 代码分割

**配置说明：** 配置是否将 CSS 代码分割成独立文件，与 JavaScript 的代码分割策略配合使用。

**配置项说明：**

- **codeSplit** - 是否启用 CSS 代码分割
  - `true` - 启用 CSS 代码分割，按路由或模块分割成独立的 CSS 文件
  - `false` - 禁用 CSS 代码分割，所有 CSS 打包到一个文件中
  - **默认值：** `true`
  - **使用场景：**
    - 启用：大型多页面应用，按需加载 CSS，减少首屏加载体积
    - 禁用：小型单页应用，减少 HTTP 请求数，简化部署

**案例 - 启用 CSS 代码分割：**

```typescript
css: {
  codeSplit: true  // 启用 CSS 代码分割，与 JS 路由同步加载
}
```

**案例 - 禁用 CSS 代码分割：**

```typescript
css: {
  codeSplit: false  // 禁用 CSS 代码分割，所有 CSS 打包到一个文件
}
```

**效果对比：**

启用 CSS 代码分割后构建输出：
```
dist/
├── index.html
├── assets/
│   ├── index.abc123.js
│   ├── index.def456.css        // 首屏 CSS
│   ├── about.ghi789.js
│   └── about.jkl012.css        // 关于页面的 CSS（按需加载）
```

禁用 CSS 代码分割后构建输出：
```
dist/
├── index.html
├── assets/
│   ├── index.abc123.js
│   └── index.def456.css        // 所有 CSS 在一个文件中
```

## 6. 静态资源处理

### 6.1 资源内联限制

**配置说明：** 配置小于指定大小的静态资源是否内联为 base64 编码，减少 HTTP 请求数。

**配置项说明：**

- **assetsInlineLimit** - 资源内联大小限制（字节）
  - 数值类型，单位为字节（byte）
  - 默认值：`4096`（4KB）
  - **使用场景：**
    - 小于此阈值的资源：内联为 base64 编码，嵌入到 JS/CSS 文件中
    - 大于此阈值的资源：作为独立文件输出，通过 URL 引用
  - **设置建议：**
    - 小图标、小图片建议内联，减少请求数
    - 大图片、视频等资源建议作为独立文件，便于缓存
    - 设置为 `0` 可以禁用资源内联，所有资源都作为独立文件输出

**案例 - 默认配置（4KB）：**

```typescript
build: {
  assetsInlineLimit: 4096  // 小于 4KB 的资源内联为 base64
}
```

**案例 - 禁用内联（所有资源独立文件）：**

```typescript
build: {
  assetsInlineLimit: 0  // 禁用内联，所有资源都作为独立文件
}
```

**案例 - 更大的内联阈值（10KB）：**

```typescript
build: {
  assetsInlineLimit: 10240  // 小于 10KB 的资源内联为 base64
}
```

**效果说明：**

资源内联后在构建输出中的表现：
```typescript
// 小于阈值的资源内联示例
const smallIcon = 'data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAA...'

// 大于阈值的资源示例
const largeImage = '/assets/large-image.abc123.png'
```

### 6.2 公共资源目录

**配置说明：** 指定静态资源目录，该目录下的文件会被原样复制到输出目录，不经过 Vite 的处理。

**配置项说明：**

- **publicDir** - 公共资源目录路径
  - 字符串类型，默认值：`'public'`
  - 可以设置为 `false` 禁用公共资源目录功能
  - **使用场景：**
    - 存放不需要经过 Vite 处理的静态资源
    - 如 favicon.ico、robots.txt、manifest.json 等
    - 如需要在 HTML 中直接通过根路径引用的资源
  - **注意事项：**
    - public 目录下的文件会原样复制到输出目录根路径
    - 在代码中引用时使用绝对路径（如 `/favicon.ico`）
    - 不要在 public 目录下存放需要被 import 的资源（如组件图片）

**案例 - 使用默认 public 目录：**

```typescript
publicDir: 'public'  // 使用默认的 public 目录
```

**案例 - 自定义公共资源目录：**

```typescript
publicDir: 'static'  // 使用 static 作为公共资源目录
```

**案例 - 禁用公共资源目录：**

```typescript
publicDir: false  // 禁用公共资源目录功能
```

**项目结构示例：**

```
project/
├── public/
│   ├── favicon.ico              // 网站图标
│   ├── robots.txt               // 爬虫规则
│   ├── manifest.json            // PWA 配置
│   └── images/
│       └── logo.png             // 直接通过 /images/logo.png 引用
├── src/
│   ├── assets/
│   │   └── component-icon.png   // 通过 import 引用，会被 Vite 处理
│   └── main.tsx
└── vite.config.ts
```

**使用示例：**

```html
<!-- index.html -->
<!DOCTYPE html>
<html>
  <head>
    <!-- 引用 public 目录下的文件 -->
    <link rel="icon" href="/favicon.ico">
    <link rel="manifest" href="/manifest.json">
  </head>
  <body>
    <!-- 引用 public 目录下的图片 -->
    <img src="/images/logo.png" alt="Logo">
  </body>
</html>
```

```typescript
// src/main.tsx
// src/assets 下的资源通过 import 引用
import componentIcon from './assets/component-icon.png'

function App() {
  return <img src={componentIcon} alt="组件图标" />
}
```
